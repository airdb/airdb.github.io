  -t deb \
  -a amd64 \
  -n test-package \
  -v 1.1.0 \
  -m 'Dean <dean@airdb.com>' \
  --url 'https://www.airdb.com' \
  -C ./build_dir/ \
  --pre-install /run.sh \
  --post-install /run.sh
